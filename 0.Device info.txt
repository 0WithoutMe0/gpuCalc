#include <sycl/sycl.hpp>
#include <iostream>

int main() {
    sycl::queue q;
    auto device = q.get_device();

    std::cout << "Device: "
        << device.get_info<sycl::info::device::name>() << std::endl;

    //глобальная память
    auto global_mem = device.get_info<sycl::info::device::global_mem_size>();
    std::cout << "Global Memory: " << global_mem / (1024 * 1024 * 1024)
        << " GB" << std::endl;

    //локальная память (на work-group)
    auto local_mem = device.get_info<sycl::info::device::local_mem_size>();
    std::cout << "Local Memory per work-group: " << local_mem / 1024
        << " KB" << std::endl;

    //максимальный размер work-group
    size_t max_wg_size = device.get_info<sycl::info::device::max_work_group_size>();
    std::cout << "Max work-group size: " << max_wg_size << " threads" << std::endl;

    //максимальное количество work-groups (по измерениям)
    auto max_compute_units = device.get_info<sycl::info::device::max_compute_units>();
    std::cout << "Max compute units: " << max_compute_units << std::endl;

    return 0;
}
