#include <sycl/sycl.hpp>
#include <iostream>
#include <chrono>
#include <vector>
#include <cassert>

const size_t data_size = 10e4;
const int time_steps = 100;
const int device_steps = 10;
const int stride = 7;

class Timer {
    std::chrono::high_resolution_clock::time_point start;
public:
    Timer() : start(std::chrono::high_resolution_clock::now()) {}
    double elapsed() {
        auto end = std::chrono::high_resolution_clock::now();
        return std::chrono::duration<double>(end - start).count();
    }
};

void init(float* data, size_t size) {
    for (size_t i = 0; i < size; i++) {
        data[i] = static_cast<float>(i % 100);
    }
}

float check(const float* data) {
    float sum = 0.0f;
    for (size_t i = 0; i < data_size; i++) {
        sum += data[i];
    }
    return sum;
}


//Serial
float serial_version() {
    float* data = new float[data_size];
    init(data, data_size);

    Timer timer;

    for (int i = 0; i < time_steps; i++) {
        //хотим вычислить на GPU
        for (size_t j = 0; j < data_size; j++) {
            for (int k = 0; k < device_steps; k++) {
                data[j] += 1.0f;
            }
        }

        //вычисление на CPU
        for (size_t j = 0; j < data_size; j += stride) 
            data[j] += 1.0f;
    }

    double elapsed = timer.elapsed();
    std::cout << "Serial time: " << elapsed << " seconds" << std::endl;

    float result = check(data);
    delete[] data;
    return result;
}

//Buffer
float buffer_version(sycl::queue& q) {
    sycl::buffer<float, 1> buffer_data{ sycl::range<1>{data_size} };
    {
        auto host_acc = buffer_data.get_host_access(sycl::write_only);
        init(host_acc.get_pointer(), data_size);
    }

    Timer timer;

    for (int i = 0; i < time_steps; i++) {
        //вычисление на GPU
        q.submit([&](sycl::handler& h) {
            auto device_acc = buffer_data.get_access<sycl::access::mode::read_write>(h);

            h.parallel_for(data_size, [=](sycl::id<1> idx) {
                for (int k = 0; k < device_steps; k++) {
                    device_acc[idx] += 1.0f;
                }
            });
        });

        //вычисление на CPU
        sycl::host_accessor host_data(buffer_data);
        for (int i = 0; i < data_size; i += stride)
            host_data[i] += 1.0;
    }

    double elapsed = timer.elapsed();
    std::cout << "Buffer time: " << elapsed << " seconds" << std::endl;

    auto host_acc = buffer_data.get_host_access(sycl::read_only);
    float result = check(host_acc.get_pointer());
    return result;
}

//Shared USM
float shared_usm_version(sycl::queue& q) {
    float* data = sycl::malloc_shared<float>(data_size, q);
    init(data, data_size);

    Timer timer;

    for (int i = 0; i < time_steps; i++) {
        //вычисление на GPU
        q.parallel_for(data_size, [=](sycl::id<1> idx) {
            for (int k = 0; k < device_steps; k++) {
                data[idx] += 1.0f;
            }
        }).wait();

        //вычисление на CPU
        for (size_t j = 0; j < data_size; j += stride)
            data[j] += 1.0f;
    }

    double elapsed = timer.elapsed();
    std::cout << "Shared USM time: " << elapsed << " seconds" << std::endl;

    float result = check(data);
    sycl::free(data, q);
    return result;
}

//Device USM
float device_usm_version(sycl::queue& q) {
    float* host_data = new float[data_size];
    init(host_data, data_size);

    float* device_data = sycl::malloc_device<float>(data_size, q);

    Timer timer;

    for (int i = 0; i < time_steps; i++) {
        //копируем данные с хоста на девайс
        auto copy1 = q.memcpy(device_data, host_data, data_size * sizeof(float));

        //вычисление на GPU
        auto kernel = q.parallel_for(data_size, copy1, [=](sycl::id<1> idx) {
            for (int k = 0; k < device_steps; k++) {
                device_data[idx] += 1.0f;
            }
        });
        
        //копируем данные с девайса на хост
        auto copy2 = q.memcpy(host_data, device_data, data_size * sizeof(float), kernel);
        copy2.wait();

        //вычисление на CPU
        for (size_t j = 0; j < data_size; j += stride)
            host_data[j] += 1.0f;
    }
    q.wait();

    double elapsed = timer.elapsed();
    std::cout << "Device USM time: " << elapsed << " seconds" << std::endl;

    float result = check(host_data);
    sycl::free(device_data, q);
    delete[] host_data;
    return result;
}



int main() {
    sycl::queue q;
    std::cout << "Running on: " << q.get_device().get_info<sycl::info::device::name>() << std::endl;

    float DESIRED = serial_version();

    float EX_1 = buffer_version(q);
    assert(DESIRED == EX_1);

    float EX_2 = shared_usm_version(q);
    assert(DESIRED == EX_2);

    float EX_3 = device_usm_version(q);
    assert(DESIRED == EX_3);


    return 0;
}
